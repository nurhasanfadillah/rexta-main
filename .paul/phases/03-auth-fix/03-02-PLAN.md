  <name>Task 1: Rewrite api/auth-proxy.ts — toNodeHandler + bodyParser:false + inline config</name>
  <files>api/auth-proxy.ts</files>
  <action>
    **Kenapa rewrite ini benar:**
    - `export const config = { api: { bodyParser: false } }` → Vercel tidak pre-parse request body, stream tetap utuh untuk `toNodeHandler`
    - Inline `betterAuth(...)` → tidak ada cross-directory ESM import (`../lib/auth`) yang gagal di Vercel runtime
    - `req.url` rewrite → `toNodeHandler` membaca `req.url` untuk menentukan action; kita set ke `/api/auth/${path}` agar Better Auth mengenali endpoint yang diminta
    - `toNodeHandler` dari `better-auth/node` → official Node.js adapter, menangani body reading + response forwarding + cookie secara internal

    **Ganti SELURUH isi `api/auth-proxy.ts` dengan:**

    ```typescript
    import { toNodeHandler } from 'better-auth/node';
    import { betterAuth } from 'better-auth';
    import { Pool } from '@neondatabase/serverless';

    export const config = { api: { bodyParser: false } };

    const pool = new Pool({
      connectionString: process.env.DATABASE_URL,
      max: 5,
    });

    const auth = betterAuth({
      database: pool,
      emailAndPassword: { enabled: true },
      secret: process.env.BETTER_AUTH_SECRET,
      baseURL: process.env.BETTER_AUTH_URL,
      trustedOrigins: [process.env.BETTER_AUTH_URL as string],
    });

    const nodeHandler = toNodeHandler(auth);

    export default function handler(req: any, res: any) {
      const p = req.query.p;
      if (p) {
        const pathStr = Array.isArray(p) ? p.join('/') : String(p);
        const decoded = decodeURIComponent(pathStr);
        const extra = Object.entries(req.query as Record<string, string>)
          .filter(([k]) => k !== 'p')
          .map(([k, v]) => `${k}=${encodeURIComponent(v)}`)
          .join('&');
        req.url = `/api/auth/${decoded}${extra ? '?' + extra : ''}`;
      }
      return nodeHandler(req, res);
    }
    ```

    **Avoid:**
    - Jangan import dari `'../lib/auth'` — ESM resolution gagal di Vercel runtime (`"type": "module"` di package.json)
    - Jangan gunakan `auth.handler(request)` dengan manual Web Request reconstruction — itu yang diganti
    - Jangan hapus atau ubah `vercel.json` — routing `/api/auth/:path*` → `/api/auth-proxy?p=*` sudah benar
    - Jangan tambah try/catch di luar `nodeHandler` — `toNodeHandler` sudah handle error internal
